00085 - (void)setMaxMinZoomScalesForCurrentBounds {
00086   CGSize imageSize = _imageView.bounds.size;
00087 
00088   // Avoid crashing if the image has no dimensions.
00089   NIDASSERT(imageSize.width > 0 && imageSize.height > 0);
00090   if (imageSize.width <= 0 || imageSize.height <= 0) {
00091     self.maximumZoomScale = 1;
00092     self.minimumZoomScale = 1;
00093     return;
00094   }
00095 
00096   // The following code is from Apple's ImageScrollView example application and has been used
00097   // here because it is well-documented and concise.
00098 
00099   CGSize boundsSize = self.bounds.size;
00100 
00101   CGFloat minScale = [self minScaleForSize:imageSize boundsSize:boundsSize];
00102 
00103   // When we show thumbnails for images that are too small for the bounds, we try to use
00104   // the known photo dimensions to scale the minimum scale to match what the final image
00105   // will be. This avoids any "snapping" effects from stretching the thumbnail too large.
00106   if ((NIPhotoScrollViewPhotoSizeThumbnail == self.photoSize)
00107       && !CGSizeEqualToSize(self.photoDimensions, CGSizeZero)) {
00108     // Modify the scale according to the final image's minScale.
00109     CGFloat minIdealScale = [self minScaleForSize:self.photoDimensions boundsSize:boundsSize];
00110     if (minIdealScale > 1) {
00111       // Only modify the scale if the final image is smaller than the photo frame.
00112       minScale = (minScale / minIdealScale);
00113     }
00114   }
00115 
00116   // On high resolution screens we have double the pixel density, so we will be seeing
00117   // every pixel if we limit the maximum zoom scale to 0.5.
00118   // If zooming disabled we always want to show the image at a 1-to-1 ratio if the image too small.
00119   // This primarily applies to the loading image on retina displays. If we use the screen scale
00120   // to calculate the max scale then the loading image will end up being half the size it should
00121   // be.
00122   CGFloat maxScale = (self.zoomingIsEnabled ? (1.0 / NIScreenScale()) : 1);
00123 
00124   if (self.photoSize != NIPhotoScrollViewPhotoSizeThumbnail) {
00125     // Don't let minScale exceed maxScale. (If the image is smaller than the screen, we
00126     // don't want to force it to be zoomed.)
00127     minScale = MIN(minScale, maxScale);
00128   }
00129 
00130   // If zooming is disabled then we flatten the range for zooming to only allow the min zoom.
00131   self.maximumZoomScale = self.zoomingIsEnabled ? maxScale : minScale;
00132   self.minimumZoomScale = minScale;
00133 }

00190 - (CGRect)rectAroundPoint:(CGPoint)point atZoomScale:(CGFloat)zoomScale {
00191   NIDASSERT(zoomScale > 0);
00192 
00193   // Define the shape of the zoom rect.
00194   CGSize boundsSize = self.bounds.size;
00195 
00196   // Modify the size according to the requested zoom level.
00197   // For example, if we're zooming in to 0.5 zoom, then this will increase the bounds size
00198   // by a factor of two.
00199   CGSize scaledBoundsSize = CGSizeMake(boundsSize.width / zoomScale,
00200                                        boundsSize.height / zoomScale);
00201 
00202   CGRect rect = CGRectMake(point.x - scaledBoundsSize.width / 2,
00203                            point.y - scaledBoundsSize.height / 2,
00204                            scaledBoundsSize.width,
00205                            scaledBoundsSize.height);
00206 
00207   // When the image is zoomed out there is a bit of empty space around the image due
00208   // to the fact that it's centered on the screen. When we created the rect around the
00209   // point we need to take this "space" into account.
00210 
00211   // 1: get the frame of the image in this view's coordinates.
00212   CGRect imageScaledFrame = [self convertRect:_imageView.frame toView:self];
00213 
00214   // 2: Offset the frame by the excess amount. This will ensure that the zoomed location
00215   //    is always centered on the tap location. We only allow positive values because a
00216   //    negative value implies that there isn't actually any offset.
00217   rect = CGRectOffset(rect, -MAX(0, imageScaledFrame.origin.x), -MAX(0, imageScaledFrame.origin.y));
00218 
00219   return rect;
00220 }

00390 - (void)restoreCenterPoint:(CGPoint)oldCenter scale:(CGFloat)oldScale {
00391   // Step 1: restore zoom scale, making sure it is within the allowable range.
00392   self.zoomScale = MIN(self.maximumZoomScale, MAX(self.minimumZoomScale, oldScale));
00393 
00394   // Step 2: restore center point, making sure it is within the allowable range.
00395 
00396   // 2a: convert our desired center point back to the scroll view's coordinate space from the
00397   //     image's coordinate space.
00398   CGPoint boundsCenter = [self convertPoint:oldCenter fromView:_imageView];
00399 
00400   // 2b: calculate the content offset that would yield that center point
00401   CGPoint offset = CGPointMake(boundsCenter.x - self.bounds.size.width / 2.0,
00402                                boundsCenter.y - self.bounds.size.height / 2.0);
00403 
00404   // 2c: restore offset, adjusted to be within the allowable range
00405   CGPoint maxOffset = [self maximumContentOffset];
00406   CGPoint minOffset = [self minimumContentOffset];
00407   offset.x = MAX(minOffset.x, MIN(maxOffset.x, offset.x));
00408   offset.y = MAX(minOffset.y, MIN(maxOffset.y, offset.y));
00409   self.contentOffset = offset;
00410 }
