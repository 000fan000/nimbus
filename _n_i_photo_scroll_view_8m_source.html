00132   // we don't want to allow the photo to be zoomed.

00135   // whatever value would make the image fit the view perfectly.

00149 - (void)setMaxMinZoomScalesForCurrentBounds {
00150   CGSize imageSize = _imageView.bounds.size;
00151 
00152   // Avoid crashing if the image has no dimensions.
00153   NIDASSERT(imageSize.width > 0 && imageSize.height > 0);
00154   if (imageSize.width <= 0 || imageSize.height <= 0) {
00155     self.maximumZoomScale = 1;
00156     self.minimumZoomScale = 1;
00157     return;
00158   }
00159 
00160   // The following code is from Apple's ImageScrollView example application and has been used
00161   // here because it is well-documented and concise.
00162 
00163   CGSize boundsSize = self.bounds.size;
00164 
00165   CGFloat minScale = 0;
00166   CGFloat maxScale = 0;
00167 
00168   // Calculate the min/max scale for the image to be presented.
00169   [self minAndMaxScaleForDimensions: imageSize
00170                          boundsSize: boundsSize
00171                           photoSize: self.photoSize
00172                            minScale: &minScale
00173                            maxScale: &maxScale];
00174 
00175   // When we show thumbnails for images that are too small for the bounds, we try to use
00176   // the known photo dimensions to scale the minimum scale to match what the final image
00177   // would be. This avoids any "snapping" effects from stretching the thumbnail too large.
00178   if ((NIPhotoScrollViewPhotoSizeThumbnail == self.photoSize)
00179       && !CGSizeEqualToSize(self.photoDimensions, CGSizeZero)) {
00180     CGFloat scaleToFitOriginal = 0;
00181     CGFloat originalMaxScale = 0;
00182     // Calculate the original-sized image's min/max scale.
00183     [self minAndMaxScaleForDimensions: self.photoDimensions
00184                            boundsSize: boundsSize
00185                             photoSize: NIPhotoScrollViewPhotoSizeOriginal
00186                              minScale: &scaleToFitOriginal
00187                              maxScale: &originalMaxScale];
00188 
00189     if (scaleToFitOriginal + FLT_EPSILON >= (1.0 / NIScreenScale())) {
00190       // If the final image will be smaller than the view then we want to use that
00191       // scale as the "true" scale and adjust it relatively to the thumbnail's dimensions.
00192       // This ensures that the thumbnail will always be the same visual size as the original
00193       // image, giving us that sexy "crisping" effect when the thumbnail is loaded.
00194       CGFloat relativeSize = self.photoDimensions.width / imageSize.width;
00195       minScale = scaleToFitOriginal * relativeSize;
00196     }
00197   }
00198 
00199   // If zooming is disabled then we flatten the range for zooming to only allow the min zoom.
00200   self.maximumZoomScale = [self isZoomingEnabled] ? maxScale : minScale;
00201   self.minimumZoomScale = minScale;
00202 }

00259 - (CGRect)rectAroundPoint:(CGPoint)point atZoomScale:(CGFloat)zoomScale {
00260   NIDASSERT(zoomScale > 0);
00261 
00262   // Define the shape of the zoom rect.
00263   CGSize boundsSize = self.bounds.size;
00264 
00265   // Modify the size according to the requested zoom level.
00266   // For example, if we're zooming in to 0.5 zoom, then this will increase the bounds size
00267   // by a factor of two.
00268   CGSize scaledBoundsSize = CGSizeMake(boundsSize.width / zoomScale,
00269                                        boundsSize.height / zoomScale);
00270 
00271   CGRect rect = CGRectMake(point.x - scaledBoundsSize.width / 2,
00272                            point.y - scaledBoundsSize.height / 2,
00273                            scaledBoundsSize.width,
00274                            scaledBoundsSize.height);
00275 
00276   // When the image is zoomed out there is a bit of empty space around the image due
00277   // to the fact that it's centered on the screen. When we created the rect around the
00278   // point we need to take this "space" into account.
00279 
00280   // 1: get the frame of the image in this view's coordinates.
00281   CGRect imageScaledFrame = [self convertRect:_imageView.frame toView:self];
00282 
00283   // 2: Offset the frame by the excess amount. This will ensure that the zoomed location
00284   //    is always centered on the tap location. We only allow positive values because a
00285   //    negative value implies that there isn't actually any offset.
00286   rect = CGRectOffset(rect, -MAX(0, imageScaledFrame.origin.x), -MAX(0, imageScaledFrame.origin.y));
00287 
00288   return rect;
00289 }

00470 - (void)restoreCenterPoint:(CGPoint)oldCenter scale:(CGFloat)oldScale {
00471   // Step 1: restore zoom scale, making sure it is within the allowable range.
00472   self.zoomScale = boundf(oldScale, self.minimumZoomScale, self.maximumZoomScale);
00473 
00474   // Step 2: restore center point, making sure it is within the allowable range.
00475 
00476   // 2a: convert our desired center point back to the scroll view's coordinate space from the
00477   //     image's coordinate space.
00478   CGPoint boundsCenter = [self convertPoint:oldCenter fromView:_imageView];
00479 
00480   // 2b: calculate the content offset that would yield that center point
00481   CGPoint offset = CGPointMake(boundsCenter.x - self.bounds.size.width / 2.0f,
00482                                boundsCenter.y - self.bounds.size.height / 2.0f);
00483 
00484   // 2c: restore offset, adjusted to be within the allowable range
00485   CGPoint maxOffset = [self maximumContentOffset];
00486   CGPoint minOffset = [self minimumContentOffset];
00487   offset.x = boundf(offset.x, minOffset.x, maxOffset.x);
00488   offset.y = boundf(offset.y, minOffset.y, maxOffset.y);
