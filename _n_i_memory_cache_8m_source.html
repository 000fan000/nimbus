00115 - (void)storeObject:(id)object withName:(NSString *)name {
00116   NIMemoryCacheInfo* info = [self cacheInfoForName:name];
00117 
00118   // Create a new cache entry.
00119   if (nil == info) {
00120     info = [[NIMemoryCacheInfo alloc] init];
00121   }
00122 
00123   // Store the object in the cache item.
00124   info.object = object;
00125 
00126   // Storing in the cache counts as an access of the object, so we update the access time.
00127   info.lastAccessTime = [NSDate date];

00129   // Clear out any expiration date that may exist so that this item won't expire.
00130   info.expirationDate = nil;

00132   // Commit the changes to the cache.

00196 - (void)reduceMemoryUsage {
00197   // Copy the cache map because it's likely that we're going to modify it.
00198   NSDictionary* cacheMap = [_cacheMap copy];
00199 
00200   // Iterate over the copied cache map (which will not be modified).
00201   for (id key in cacheMap) {
00202     NIMemoryCacheInfo* info = [_cacheMap objectForKey:key];

00204     if ([info hasExpired]) {
00205       [_cacheMap removeObjectForKey:key];
