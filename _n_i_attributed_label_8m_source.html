00391     NSError* error = nil;

00393                                                                    error:&error];

00395     NSRange range = NSMakeRange(0, string.length);
00396 
00397     [_detectedlinkLocations release];
00398     _detectedlinkLocations = [[linkDetector matchesInString:string options:0 range:range] retain];
00399

00446 - (NSTextCheckingResult *)linkAtPoint:(CGPoint)point {
00447   static const CGFloat kVMargin = 5.0f;
00448   if (!CGRectContainsPoint(CGRectInset(self.bounds, 0, -kVMargin), point)) {
00449     return nil;
00450   }
00451 
00452   CFArrayRef lines = CTFrameGetLines(_textFrame);
00453   if (!lines) return nil;
00454   CFIndex count = CFArrayGetCount(lines);
00455 
00456   NSTextCheckingResult* foundLink = nil;
00457 
00458   CGPoint origins[count];
00459   CTFrameGetLineOrigins(_textFrame, CFRangeMake(0,0), origins);
00460 
00461   for (int i = 0; i < count; i++) {
00462     CGPoint linePoint = origins[i];
00463 
00464     CTLineRef line = CFArrayGetValueAtIndex(lines, i);
00465     CGRect flippedRect = [self getLineBounds:line point:linePoint];
00466     CGRect bounds = CGRectMake(CGRectGetMinX(self.bounds),
00467                                CGRectGetMaxY(self.bounds)-CGRectGetMaxY(self.bounds),
00468                                CGRectGetWidth(self.bounds),
00469                                CGRectGetHeight(self.bounds));
00470     CGRect rect = CGRectMake(CGRectGetMinX(flippedRect),
00471                              CGRectGetMaxY(bounds)-CGRectGetMaxY(flippedRect),
00472                              CGRectGetWidth(flippedRect),
00473                              CGRectGetHeight(flippedRect));                      
00474 
00475     rect = CGRectInset(rect, 0, -kVMargin);
00476     if (CGRectContainsPoint(rect, point)) {
00477       CGPoint relativePoint = CGPointMake(point.x-CGRectGetMinX(rect),
00478                                           point.y-CGRectGetMinY(rect));
00479       CFIndex idx = CTLineGetStringIndexForPosition(line, relativePoint);
00480       foundLink = ([self linkAtIndex:idx]);
00481       if (foundLink) return foundLink;
00482     }
00483   }
00484   return nil;
00485 }

00564 - (void)drawTextInRect:(CGRect)rect {
00565   if (self.autoDetectLinks) {
00566     [self detectLinks];
00567   }
00568 
00569   NSMutableAttributedString* attributedStringWithLinks =
00570     [self mutableAttributedStringWithLinkStylesApplied];
00571   self.userInteractionEnabled = (_detectedlinkLocations.count > 0
00572                                  || _explicitLinkLocations.count > 0);
00573 
00574   if (nil != attributedStringWithLinks) {
00575     CGContextRef ctx = UIGraphicsGetCurrentContext();
00576     CGContextSaveGState(ctx);
00577 
00578     // CoreText context coordinates are the opposite to UIKit so we flip the bounds
00579     CGContextConcatCTM(ctx, CGAffineTransformScale(CGAffineTransformMakeTranslation(0, self.bounds.size.height),
00580                                                    1.f, -1.f));
00581 
00582     if (nil == _textFrame) {
00583       CFAttributedStringRef attributedString = (CFAttributedStringRef)attributedStringWithLinks;
00584       CTFramesetterRef framesetter = CTFramesetterCreateWithAttributedString(attributedString);
00585 
00586       CGMutablePathRef path = CGPathCreateMutable();
00587       CGPathAddRect(path, NULL, self.bounds);
00588       _textFrame = CTFramesetterCreateFrame(framesetter, CFRangeMake(0, 0), path, NULL);
00589       CGPathRelease(path);
00590       CFRelease(framesetter);
00591     }
00592 
00593     if (nil != _touchedLink) {
00594       // Draw the link's background first.
00595       [self.linkHighlightColor setFill];
00596 
00597       NSRange linkRange = _touchedLink.range;
00598 
00599       CFArrayRef lines = CTFrameGetLines(_textFrame);
00600       CFIndex count = CFArrayGetCount(lines);
00601       CGPoint lineOrigins[count];
00602       CTFrameGetLineOrigins(_textFrame, CFRangeMake(0, 0), lineOrigins);
00603 
00604       for (CFIndex i = 0; i < count; i++) {
00605         CTLineRef line = CFArrayGetValueAtIndex(lines, i);
00606 
00607         CFRange stringRange = CTLineGetStringRange(line);
00608         NSRange lineRange = NSMakeRange(stringRange.location, stringRange.length);
00609         NSRange intersectedRange = NSIntersectionRange(lineRange, linkRange);
00610         if (intersectedRange.length == 0) {
00611           continue;
00612         }
00613 
00614         // Iterate through each of the "runs" (i.e. a chunk of text) and find the runs that
00615         // intersect with the link's range. For each of these runs, draw the highlight frame
00616         // around the part of the run that intersects with the link.
00617         CGRect highlightRect = CGRectZero;
00618         CFArrayRef runs = CTLineGetGlyphRuns(line);
00619         CFIndex runCount = CFArrayGetCount(runs);
00620         for (CFIndex k = 0; k < runCount; k++) {
00621           CTRunRef run = CFArrayGetValueAtIndex(runs, k);
00622 
00623           CFRange stringRunRange = CTRunGetStringRange(run);
00624           NSRange lineRunRange = NSMakeRange(stringRunRange.location, stringRunRange.length);
00625           NSRange intersectedRunRange = NSIntersectionRange(lineRunRange, linkRange);
00626           if (intersectedRunRange.length == 0) {
00627             continue;
00628           }
00629 
00630           CGFloat ascent = 0.0f;
00631           CGFloat descent = 0.0f;
00632           CGFloat leading = 0.0f;
00633           CGFloat width = (CGFloat)CTRunGetTypographicBounds(run,
00634                                                              CFRangeMake(0, 0), 
00635                                                              &ascent, 
00636                                                              &descent, 
00637                                                              &leading);
00638           CGFloat height = ascent + descent;
00639 
00640           CGFloat xOffset = CTLineGetOffsetForStringIndex(line, 
00641                                                           CTRunGetStringRange(run).location, 
00642                                                           nil);
00643 
00644           CGRect linkRect = CGRectMake(lineOrigins[i].x + xOffset - leading,
00645                                        lineOrigins[i].y - descent,
00646                                        width + leading,
00647                                        height);
00648 
00649           linkRect = CGRectIntegral(linkRect);
00650           linkRect = CGRectInset(linkRect, -2, -1);
00651 
00652           if (CGRectIsEmpty(highlightRect)) {
00653             highlightRect = linkRect;
00654 
00655           } else {
00656             highlightRect = CGRectUnion(highlightRect, linkRect);
00657           }
00658         }
00659 
00660         if (!CGRectIsEmpty(highlightRect)) {
00661 
00662           CGFloat pi = (CGFloat)M_PI;
00663 
00664           CGFloat radius = 5.0f;
00665           CGContextMoveToPoint(ctx, highlightRect.origin.x, highlightRect.origin.y + radius);
00666           CGContextAddLineToPoint(ctx, highlightRect.origin.x, highlightRect.origin.y + highlightRect.size.height - radius);
00667           CGContextAddArc(ctx, highlightRect.origin.x + radius, highlightRect.origin.y + highlightRect.size.height - radius, 
00668                           radius, pi, pi / 2.0f, 1.0f);
00669           CGContextAddLineToPoint(ctx, highlightRect.origin.x + highlightRect.size.width - radius, 
00670                                   highlightRect.origin.y + highlightRect.size.height);
00671           CGContextAddArc(ctx, highlightRect.origin.x + highlightRect.size.width - radius, 
00672                           highlightRect.origin.y + highlightRect.size.height - radius, radius, pi / 2, 0.0f, 1.0f);
00673           CGContextAddLineToPoint(ctx, highlightRect.origin.x + highlightRect.size.width, highlightRect.origin.y + radius);
00674           CGContextAddArc(ctx, highlightRect.origin.x + highlightRect.size.width - radius, highlightRect.origin.y + radius, 
00675                           radius, 0.0f, -pi / 2.0f, 1.0f);
00676           CGContextAddLineToPoint(ctx, highlightRect.origin.x + radius, highlightRect.origin.y);
00677           CGContextAddArc(ctx, highlightRect.origin.x + radius, highlightRect.origin.y + radius, radius, 
00678                           -pi / 2, pi, 1);
00679           CGContextFillPath(ctx);
00680         }
00681       }
00682     }
00683 
00684     CTFrameDraw(_textFrame, ctx);
00685     CGContextRestoreGState(ctx);
00686 
00687   } else {
00688     [super drawTextInRect:rect];
00689   }
00690 }
