00422 - (NSTextCheckingResult *)linkAtPoint:(CGPoint)point {
00423   static const CGFloat kVMargin = 5.0f;
00424   if (!CGRectContainsPoint(CGRectInset(self.bounds, 0, -kVMargin), point)) {
00425     return nil;
00426   }
00427 
00428   CFArrayRef lines = CTFrameGetLines(_textFrame);
00429   if (!lines) return nil;
00430   CFIndex count = CFArrayGetCount(lines);
00431 
00432   NSTextCheckingResult* foundLink = nil;
00433 
00434   CGPoint origins[count];
00435   CTFrameGetLineOrigins(_textFrame, CFRangeMake(0,0), origins);
00436 
00437   for (int i = 0; i < count; i++) {
00438     CGPoint linePoint = origins[i];
00439 
00440     CTLineRef line = CFArrayGetValueAtIndex(lines, i);
00441     CGRect flippedRect = [self getLineBounds:line point:linePoint];
00442     CGRect bounds = CGRectMake(CGRectGetMinX(self.bounds),
00443                                CGRectGetMaxY(self.bounds)-CGRectGetMaxY(self.bounds),
00444                                CGRectGetWidth(self.bounds),
00445                                CGRectGetHeight(self.bounds));
00446     CGRect rect = CGRectMake(CGRectGetMinX(flippedRect),
00447                              CGRectGetMaxY(bounds)-CGRectGetMaxY(flippedRect),
00448                              CGRectGetWidth(flippedRect),
00449                              CGRectGetHeight(flippedRect));                      
00450 
00451     rect = CGRectInset(rect, 0, -kVMargin);
00452     if (CGRectContainsPoint(rect, point)) {
00453       CGPoint relativePoint = CGPointMake(point.x-CGRectGetMinX(rect),
00454                                           point.y-CGRectGetMinY(rect));
00455       CFIndex idx = CTLineGetStringIndexForPosition(line, relativePoint);
00456       foundLink = ([self linkAtIndex:idx]);
00457       if (foundLink) return foundLink;
00458     }
00459   }
00460   return nil;
00461 }

00540 - (void)drawTextInRect:(CGRect)rect {
00541   if (self.autoDetectLinks) {
00542     [self detectLinks];
00543   }
00544 
00545   NSMutableAttributedString* attributedStringWithLinks =
00546     [self mutableAttributedStringWithLinkStylesApplied];
00547   self.userInteractionEnabled = (_detectedlinkLocations.count > 0
00548                                  || _explicitLinkLocations.count > 0);
00549 
00550   if (nil != attributedStringWithLinks) {
00551     CGContextRef ctx = UIGraphicsGetCurrentContext();
00552     CGContextSaveGState(ctx);
00553 
00554     // CoreText context coordinates are the opposite to UIKit so we flip the bounds
00555     CGContextConcatCTM(ctx, CGAffineTransformScale(CGAffineTransformMakeTranslation(0, self.bounds.size.height),
00556                                                    1.f, -1.f));
00557 
00558     if (nil == _textFrame) {
00559       CFAttributedStringRef attributedString = (CFAttributedStringRef)attributedStringWithLinks;
00560       CTFramesetterRef framesetter = CTFramesetterCreateWithAttributedString(attributedString);
00561 
00562       CGMutablePathRef path = CGPathCreateMutable();
00563       CGPathAddRect(path, NULL, self.bounds);
00564       _textFrame = CTFramesetterCreateFrame(framesetter, CFRangeMake(0, 0), path, NULL);
00565       CGPathRelease(path);
00566       CFRelease(framesetter);
00567     }
00568 
00569     if (nil != _touchedLink) {
00570       // Draw the link's background first.
00571       [self.linkHighlightColor setFill];
00572 
00573       NSRange linkRange = _touchedLink.range;
00574 
00575       CFArrayRef lines = CTFrameGetLines(_textFrame);
00576       CFIndex count = CFArrayGetCount(lines);
00577       CGPoint lineOrigins[count];
00578       CTFrameGetLineOrigins(_textFrame, CFRangeMake(0, 0), lineOrigins);
00579 
00580       for (CFIndex i = 0; i < count; i++) {
00581         CTLineRef line = CFArrayGetValueAtIndex(lines, i);
00582 
00583         CFRange stringRange = CTLineGetStringRange(line);
00584         NSRange lineRange = NSMakeRange(stringRange.location, stringRange.length);
00585         NSRange intersectedRange = NSIntersectionRange(lineRange, linkRange);
00586         if (intersectedRange.length == 0) {
00587           continue;
00588         }
00589 
00590         // Iterate through each of the "runs" (i.e. a chunk of text) and find the runs that
00591         // intersect with the link's range. For each of these runs, draw the highlight frame
00592         // around the part of the run that intersects with the link.
00593         CGRect highlightRect = CGRectZero;
00594         CFArrayRef runs = CTLineGetGlyphRuns(line);
00595         CFIndex runCount = CFArrayGetCount(runs);
00596         for (CFIndex k = 0; k < runCount; k++) {
00597           CTRunRef run = CFArrayGetValueAtIndex(runs, k);
00598 
00599           CFRange stringRunRange = CTRunGetStringRange(run);
00600           NSRange lineRunRange = NSMakeRange(stringRunRange.location, stringRunRange.length);
00601           NSRange intersectedRunRange = NSIntersectionRange(lineRunRange, linkRange);
00602           if (intersectedRunRange.length == 0) {
00603             continue;
00604           }
00605 
00606           CGFloat ascent = 0.0f;
00607           CGFloat descent = 0.0f;
00608           CGFloat leading = 0.0f;
00609           CGFloat width = (CGFloat)CTRunGetTypographicBounds(run,
00610                                                              CFRangeMake(0, 0), 
00611                                                              &ascent, 
00612                                                              &descent, 
00613                                                              &leading);
00614           CGFloat height = ascent + descent;
00615 
00616           CGFloat xOffset = CTLineGetOffsetForStringIndex(line, 
00617                                                           CTRunGetStringRange(run).location, 
00618                                                           nil);
00619 
00620           CGRect linkRect = CGRectMake(lineOrigins[i].x + xOffset - leading,
00621                                        lineOrigins[i].y - descent,
00622                                        width + leading,
00623                                        height);
00624 
00625           linkRect = CGRectIntegral(linkRect);
00626           linkRect = CGRectInset(linkRect, -2, -1);
00627 
00628           if (CGRectIsEmpty(highlightRect)) {
00629             highlightRect = linkRect;
00630 
00631           } else {
00632             highlightRect = CGRectUnion(rect, linkRect);
00633           }
00634         }
00635 
00636         if (!CGRectIsEmpty(highlightRect)) {
00637 
00638           CGFloat pi = (CGFloat)M_PI;
00639 
00640           CGFloat radius = 5.0f;
00641           CGContextMoveToPoint(ctx, highlightRect.origin.x, highlightRect.origin.y + radius);
00642           CGContextAddLineToPoint(ctx, highlightRect.origin.x, highlightRect.origin.y + highlightRect.size.height - radius);
00643           CGContextAddArc(ctx, highlightRect.origin.x + radius, highlightRect.origin.y + highlightRect.size.height - radius, 
00644                           radius, pi, pi / 2.0f, 1.0f);
00645           CGContextAddLineToPoint(ctx, highlightRect.origin.x + highlightRect.size.width - radius, 
00646                                   highlightRect.origin.y + highlightRect.size.height);
00647           CGContextAddArc(ctx, highlightRect.origin.x + highlightRect.size.width - radius, 
00648                           highlightRect.origin.y + highlightRect.size.height - radius, radius, pi / 2, 0.0f, 1.0f);
00649           CGContextAddLineToPoint(ctx, highlightRect.origin.x + highlightRect.size.width, highlightRect.origin.y + radius);
00650           CGContextAddArc(ctx, highlightRect.origin.x + highlightRect.size.width - radius, highlightRect.origin.y + radius, 
00651                           radius, 0.0f, -pi / 2.0f, 1.0f);
00652           CGContextAddLineToPoint(ctx, highlightRect.origin.x + radius, highlightRect.origin.y);
00653           CGContextAddArc(ctx, highlightRect.origin.x + radius, highlightRect.origin.y + radius, radius, 
00654                           -pi / 2, pi, 1);
00655           CGContextFillPath(ctx);
00656         }
00657       }
00658     }
00659 
00660     CTFrameDraw(_textFrame, ctx);
00661     CGContextRestoreGState(ctx);
00662 
00663   } else {
00664     [super drawTextInRect:rect];
00665   }
00666 }
