00458   if (_attributedText) {
00459     CGContextRef ctx = UIGraphicsGetCurrentContext();
00460     CGContextSaveGState(ctx);
00461

00463       [self linksDetectedAttributedString] : [[self.attributedText copy] autorelease];
00464 
00465     // CoreText context coordinates are the opposite to UIKit so we flip the bounds
00466     CGContextConcatCTM(ctx,
00467                        CGAffineTransformScale(CGAffineTransformMakeTranslation(0, self.bounds.size.height),
00468                                               1.f, -1.f));
00469     
00470     if (_textFrame == nil) {
00471       CTFramesetterRef framesetter =
00472         CTFramesetterCreateWithAttributedString((CFAttributedStringRef)attributedString);
00473       _drawingRect = self.bounds;
00474       CGMutablePathRef path = CGPathCreateMutable();
00475       CGPathAddRect(path, NULL, _drawingRect);
00476       _textFrame = CTFramesetterCreateFrame(framesetter,CFRangeMake(0,0), path, NULL);
00477       CGPathRelease(path);
00478       CFRelease(framesetter);
00479     }
00480     
00481     if (_currentLink) {
00482       
00483       [self.linkHighlightColor setFill];
00484       
00485       NSRange linkRange = _currentLink.range;
00486       
00487       CFArrayRef lines = CTFrameGetLines(_textFrame);
00488       CFIndex count = CFArrayGetCount(lines);
00489       CGPoint lineOrigins[count];
00490       CTFrameGetLineOrigins(_textFrame, CFRangeMake(0, 0), lineOrigins);
00491       
00492       for (CFIndex i = 0; i < count; i++) {
00493         CTLineRef line = CFArrayGetValueAtIndex(lines, i);
00494         
00495         CFRange stringRange = CTLineGetStringRange(line);
00496         NSRange lineRange = NSMakeRange(stringRange.location, stringRange.length);
00497         NSRange intersectedRange = NSIntersectionRange(lineRange, linkRange);
00498         if (intersectedRange.length == 0) {
00499           continue;
00500         }
00501         
00502         CGRect highlightRect = CGRectZero;
00503         CFArrayRef runs = CTLineGetGlyphRuns(line);
00504         CFIndex runCount = CFArrayGetCount(runs);
00505         for (CFIndex k = 0; k < runCount; k++) {
00506           CTRunRef run = CFArrayGetValueAtIndex(runs, k);
00507           
00508           CFRange stringRunRange = CTRunGetStringRange(run);
00509           NSRange lineRunRange = NSMakeRange(stringRunRange.location, stringRunRange.length);
00510           NSRange intersectedRunRange = NSIntersectionRange(lineRunRange, linkRange);
00511           if (intersectedRunRange.length == 0) {
00512             continue;
00513           }
00514 
00515           CGFloat ascent = 0.0f;
00516           CGFloat descent = 0.0f;
00517           CGFloat leading = 0.0f;
00518           CGFloat width = (CGFloat)CTRunGetTypographicBounds(run, 
00519                                                     CFRangeMake(0, 0), 
00520                                                     &ascent, 
00521                                                     &descent, 
00522                                                     &leading);
00523           CGFloat height = ascent + descent;
00524           
00525           CGFloat xOffset = CTLineGetOffsetForStringIndex(line, 
00526                                                           CTRunGetStringRange(run).location, 
00527                                                           nil);
00528           
00529           CGRect linkRect = CGRectMake(lineOrigins[i].x + xOffset - leading,
00530                             lineOrigins[i].y - descent,
00531                             width + leading,
00532                             height);
00533           
00534           linkRect = CGRectIntegral(linkRect);

00536           if (CGRectIsEmpty(highlightRect)) {
00537             highlightRect = linkRect;
00538           } else {
00539             highlightRect = CGRectUnion(rect, linkRect);
00540           }
00541         }

00550           CGContextAddArc(ctx, highlightRect.origin.x + radius, highlightRect.origin.y + highlightRect.size.height - radius, 
00551                           radius, pi, pi / 2.0f, 1.0f);
00552           CGContextAddLineToPoint(ctx, highlightRect.origin.x + highlightRect.size.width - radius, 
00553                                   highlightRect.origin.y + highlightRect.size.height);
00554           CGContextAddArc(ctx, highlightRect.origin.x + highlightRect.size.width - radius, 
00555                           highlightRect.origin.y + highlightRect.size.height - radius, radius, pi / 2, 0.0f, 1.0f);
00556           CGContextAddLineToPoint(ctx, highlightRect.origin.x + highlightRect.size.width, highlightRect.origin.y + radius);
00557           CGContextAddArc(ctx, highlightRect.origin.x + highlightRect.size.width - radius, highlightRect.origin.y + radius, 
00558                           radius, 0.0f, -pi / 2.0f, 1.0f);
00559           CGContextAddLineToPoint(ctx, highlightRect.origin.x + radius, highlightRect.origin.y);
00560           CGContextAddArc(ctx, highlightRect.origin.x + radius, highlightRect.origin.y + radius, radius, 
00561                           -pi / 2, pi, 1);
