00091 - (void)shareAction {
00092   // Dismiss the action menu if the user taps the action button again on the iPad.
00093   if ([_actionSheet isVisible]) {
00094     // It shouldn't be possible to tap the share action button again on anything but the iPad.
00095     NIDASSERT(NIIsPad());
00096 
00097     [_actionSheet dismissWithClickedButtonIndex:[_actionSheet cancelButtonIndex] animated:YES];
00098 
00099     // We remove the action sheet here just in case the delegate isn't properly implemented.
00100     _actionSheet.delegate = nil;
00101     NI_RELEASE_SAFELY(_actionSheet);
00102     NI_RELEASE_SAFELY(_actionSheetURL);
00103 
00104     // Don't show the menu again.
00105     return;
00106   }
00107 
00108   // Remember the URL at this point
00109   [_actionSheetURL release];
00110   _actionSheetURL = [self.URL copy];
00111 
00112   if (nil == _actionSheet) {
00113     _actionSheet =
00114     [[UIActionSheet alloc] initWithTitle:[_actionSheetURL absoluteString]
00115                                 delegate:self
00116                        cancelButtonTitle:nil
00117                   destructiveButtonTitle:nil
00118                        otherButtonTitles:nil];
00119     // Let -shouldPresentActionSheet: setup the action sheet
00120     if (![self shouldPresentActionSheet:_actionSheet]) {
00121       // A subclass decided to handle the action in another way
00122       NI_RELEASE_SAFELY(_actionSheet);
00123       NI_RELEASE_SAFELY(_actionSheetURL);
00124       return;
00125     }
00126     // Add "Cancel" button except for iPads
00127     if (!NIIsPad()) {
00128       [_actionSheet setCancelButtonIndex:[_actionSheet addButtonWithTitle:NSLocalizedString(@"Cancel", @"")]];
00129     }
00130   }
00131 
00132   if (NIIsPad()) {
00133     [_actionSheet showFromBarButtonItem:_actionButton animated:YES];
00134   } else {
00135     [_actionSheet showInView:self.view];
00136   }
