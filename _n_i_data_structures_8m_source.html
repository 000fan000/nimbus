00255 - (NSUInteger)countByEnumeratingWithState: (NSFastEnumerationState *)state
00256                                   objects: (id *)stackbuf
00257                                     count: (NSUInteger)len {
00258   // Initialization condition.
00259   if (0 == state->state) {
00260     // Whenever the linked list is modified, the modification number increases. This allows
00261     // enumeration to bail out if the linked list is modified mid-flight.
00262     state->mutationsPtr = &_modificationNumber;
00263   }
00264 
00265   NSUInteger numberOfItemsReturned = 0;
00266 
00267   // If there is no _tail (i.e. this is an empty list) then this will end immediately.
00268   if ((struct NILinkedListNode *)state->state != _tail) {
00269     state->itemsPtr = stackbuf;
00270 
00271     if (0 == state->state) {
00272       // Initialize the state here instead of above when we check 0 == state->state because
00273       // for single item linked lists head == tail. If we initialized it in the initialization
00274       // condition, state->state != _tail check would fail and we wouldn't return the single
00275       // object.
00276       state->state = (unsigned long)_head;
00277     }
00278 
00279     // Return *at most* the number of request objects.
00280     while ((0 != state->state) && (numberOfItemsReturned < len)) {
00281       struct NILinkedListNode* node = (struct NILinkedListNode *)state->state;
00282       stackbuf[numberOfItemsReturned] = node->object;
00283       state->state = (unsigned long)node->next;
00284       ++numberOfItemsReturned;
00285     }
00286 
00287     if (0 == state->state) {
00288       // Final step condition. We allow the above loop to overstep the end one iteration,
00289       // because we rewind it one step here (to ensure that the next time enumeration occurs,
00290       // state == _tail.
00291       state->state = (unsigned long)_tail;
00292     }
00293 
00294   } // else we've returned all of the items that we can; leave numberOfItemsReturned as 0 to
00295     // signal that there is nothing left to be done.
00296 
00297   return numberOfItemsReturned;
00298 }
