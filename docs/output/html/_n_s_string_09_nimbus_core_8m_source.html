00076   NSMutableDictionary* pairs = [NSMutableDictionary dictionary];

00078   while (![scanner isAtEnd]) {
00079     NSString* pairString = nil;
00080     [scanner scanUpToCharactersFromSet:delimiterSet intoString:&pairString];
00081     [scanner scanCharactersFromSet:delimiterSet intoString:NULL];
00082     NSArray* kvPair = [pairString componentsSeparatedByString:@"="];
00083     if (kvPair.count == 1 || kvPair.count == 2) {
00084       NSString* key = [[kvPair objectAtIndex:0]
00085                        stringByReplacingPercentEscapesUsingEncoding:encoding];
00086       NSMutableArray* values = [pairs objectForKey:key];
00087       if (nil == values) {
00088         values = [NSMutableArray array];
00089         [pairs setObject:values forKey:key];
00090       }
00091       if (kvPair.count == 1) {
00092         [values addObject:[NSNull null]];
00093 
00094       } else if (kvPair.count == 2) {
00095         NSString* value = [[kvPair objectAtIndex:1]
00096                            stringByReplacingPercentEscapesUsingEncoding:encoding];
00097         [values addObject:value];
00098       }
00099     }
00100   }

00166 - (NSComparisonResult)versionStringCompare:(NSString *)other {
00167   NSArray *oneComponents = [self componentsSeparatedByString:@"a"];
00168   NSArray *twoComponents = [other componentsSeparatedByString:@"a"];
00169 
00170   // The parts before the "a"
00171   NSString *oneMain = [oneComponents objectAtIndex:0];
00172   NSString *twoMain = [twoComponents objectAtIndex:0];
00173 
00174   // If main parts are different, return that result, regardless of alpha part
00175   NSComparisonResult mainDiff;
00176   if ((mainDiff = [oneMain compare:twoMain]) != NSOrderedSame) {
00177     return mainDiff;
00178   }
00179 
00180   // At this point the main parts are the same; just deal with alpha stuff
00181   // If one has an alpha part and the other doesn't, the one without is newer
00182   if ([oneComponents count] < [twoComponents count]) {
00183     return NSOrderedDescending;
00184 
00185   } else if ([oneComponents count] > [twoComponents count]) {
00186     return NSOrderedAscending;
00187 
00188   } else if ([oneComponents count] == 1) {
00189     // Neither has an alpha part, and we know the main parts are the same
00190     return NSOrderedSame;
00191   }
00192 
00193   // At this point the main parts are the same and both have alpha parts. Compare the alpha parts
00194   // numerically. If it's not a valid number (including empty string) it's treated as zero.
00195   NSNumber *oneAlpha = [NSNumber numberWithInt:[[oneComponents objectAtIndex:1] intValue]];
00196   NSNumber *twoAlpha = [NSNumber numberWithInt:[[twoComponents objectAtIndex:1] intValue]];
00197   return [oneAlpha compare:twoAlpha];
