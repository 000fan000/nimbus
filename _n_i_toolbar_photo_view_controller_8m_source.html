00309 - (void)setChromeVisibility:(BOOL)isVisible animated:(BOOL)animated {
00310   if (_isAnimatingChrome
00311       || (!isVisible && self.toolbar.hidden)
00312       || (isVisible && !self.toolbar.hidden)
00313       || !self.chromeCanBeHidden) {
00314     // Nothing to do here.
00315     return;
00316   }
00317 
00318   CGRect toolbarFrame = self.toolbar.frame;
00319   CGRect bounds = self.view.bounds;
00320 
00321   // Reset the toolbar's initial position.
00322   if (!isVisible) {
00323     toolbarFrame.origin.y = bounds.size.height - toolbarFrame.size.height;
00324 
00325   } else {
00326     // Ensure that the toolbar is visible through the animation.
00327     self.toolbar.hidden = NO;
00328 
00329     toolbarFrame.origin.y = bounds.size.height;
00330   }
00331   self.toolbar.frame = toolbarFrame;
00332 
00333   // Show/hide the system chrome.
00334   if ([[UIApplication sharedApplication] respondsToSelector:
00335        @selector(setStatusBarHidden:withAnimation:)]) {
00336     // On 3.2 and higher we can slide the status bar out.
00337     [[UIApplication sharedApplication] setStatusBarHidden: !isVisible
00338                                             withAnimation: (animated
00339                                                             ? UIStatusBarAnimationSlide
00340                                                             : UIStatusBarAnimationNone)];
00341 
00342   } else {
00343 #if __IPHONE_OS_VERSION_MIN_REQUIRED < NIIOS_3_2
00344     // On 3.0 devices we use the boring fade animation.
00345     [[UIApplication sharedApplication] setStatusBarHidden: !isVisible
00346                                                  animated: animated];
00347 #endif
00348   }
00349 
00350   // Place the toolbar at its final location.
00351   if (isVisible) {
00352     // Slide up.
00353     toolbarFrame.origin.y = bounds.size.height - toolbarFrame.size.height;
00354 
00355   } else {
00356     // Slide down.
00357     toolbarFrame.origin.y = bounds.size.height;
00358   }
00359 
00360   // If there is a navigation bar, place it at its final location.
00361   CGRect navigationBarFrame = CGRectZero;
00362   if (nil != self.navigationController.navigationBar) {
00363     navigationBarFrame = self.navigationController.navigationBar.frame;
00364     CGRect statusBarFrame = [[UIApplication sharedApplication] statusBarFrame];
00365     CGFloat statusBarHeight = MIN(statusBarFrame.size.width, statusBarFrame.size.height);
00366 
00367     if (isVisible) {
00368       navigationBarFrame.origin.y = statusBarHeight;
00369 
00370     } else {
00371       navigationBarFrame.origin.y = 0;
00372     }
00373   }
00374 
00375   if (animated) {
00376     [UIView beginAnimations:nil context:nil];
00377     [UIView setAnimationDelegate:self];
00378     [UIView setAnimationDidStopSelector:(isVisible
00379                                          ? @selector(didShowChrome)
00380                                          : @selector(didHideChrome))];
00381 
00382     // Ensure that the animation matches the status bar's.
00383     [UIView setAnimationDuration:NIStatusBarAnimationDuration()];
00384     [UIView setAnimationCurve:NIStatusBarAnimationCurve()];

00389     self.navigationController.navigationBar.frame = navigationBarFrame;

00419 - (void)didTap {
00420   SEL selector = @selector(toggleChromeVisibility);
00421   if (self.photoAlbumView.zoomingIsEnabled) {
00422     // Cancel any previous delayed performs so that we don't stack them.
00423     [NSObject cancelPreviousPerformRequestsWithTarget:self selector:selector object:nil];
00424 
00425     // We need to delay taking action on the first tap in case a second tap comes in, causing
00426     // a double-tap gesture to be recognized and the photo to be zoomed.
00427     [self performSelector: selector
00428                withObject: nil
00429                afterDelay: 0.3];
00430 
00431   } else {
00432     // When zooming is disabled, double-tap-to-zoom is also disabled so we don't have to
00433     // be as careful; just toggle the chrome immediately.
00434     [self toggleChromeVisibility];
00435   }
00436 }
