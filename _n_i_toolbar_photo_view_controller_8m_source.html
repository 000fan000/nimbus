00265 - (void)setChromeVisibility:(BOOL)isVisible animated:(BOOL)animated {
00266   if (_isAnimatingChrome
00267       || (!isVisible && self.toolbar.hidden)
00268       || (isVisible && !self.toolbar.hidden)
00269       || !self.chromeCanBeHidden) {
00270     // Nothing to do here.
00271     return;

00275   CGRect bounds = self.view.bounds;
00276 
00277   // Reset the toolbar's initial position.
00278   if (!isVisible) {
00279     toolbarFrame.origin.y = bounds.size.height - toolbarFrame.size.height;
00280 
00281   } else {
00282     // Ensure that the toolbar is visible through the animation.
00283     self.toolbar.hidden = NO;
00284 
00285     toolbarFrame.origin.y = bounds.size.height;
00286   }
00287   self.toolbar.frame = toolbarFrame;
00288 
00289   // Show/hide the system chrome.
00290   if ([[UIApplication sharedApplication] respondsToSelector:
00291        @selector(setStatusBarHidden:withAnimation:)]) {
00292     // On 3.2 and higher we can slide the status bar out.
00293     [[UIApplication sharedApplication] setStatusBarHidden: !isVisible
00294                                             withAnimation: (animated
00295                                                             ? UIStatusBarAnimationSlide
00296                                                             : UIStatusBarAnimationNone)];
00297 
00298   } else {
00299     // On 3.0 devices we use the boring fade animation.
00300     [[UIApplication sharedApplication] setStatusBarHidden: !isVisible
00301                                                  animated: animated];
00302   }
00303 
00304   // Place the toolbar at its final location.
00305   if (isVisible) {
00306     // Slide up.
00307     toolbarFrame.origin.y = bounds.size.height - toolbarFrame.size.height;
00308 
00309   } else {
00310     // Slide down.
00311     toolbarFrame.origin.y = bounds.size.height;
00312   }
00313 
00314   // If there is a navigation bar, place it at its final location.
00315   CGRect navigationBarFrame = CGRectZero;
00316   if (nil != self.navigationController.navigationBar) {
00317     navigationBarFrame = self.navigationController.navigationBar.frame;
00318     CGRect statusBarFrame = [[UIApplication sharedApplication] statusBarFrame];
00319     CGFloat statusBarHeight = MIN(statusBarFrame.size.width, statusBarFrame.size.height);
00320 
00321     if (isVisible) {
00322       navigationBarFrame.origin.y = statusBarHeight;
00323 
00324     } else {
00325       navigationBarFrame.origin.y = 0;
00326     }
00327   }
00328 
00329   if (animated) {
00330     [UIView beginAnimations:nil context:nil];

00333                                          ? @selector(didShowChrome)
00334                                          : @selector(didHideChrome))];
00335

00375   if (self.photoAlbumView.zoomingIsEnabled) {
00376     // Cancel any previous delayed performs so that we don't stack them.
