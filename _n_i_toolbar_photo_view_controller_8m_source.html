00338   if (self.photoAlbumView.zoomingIsEnabled) {
00339     // Cancel any previous delayed performs so that we don't stack them.
00340     [NSObject cancelPreviousPerformRequestsWithTarget:self selector:selector object:nil];
00341 
00342     // We need to delay taking action on the first tap in case a second tap comes in, causing
