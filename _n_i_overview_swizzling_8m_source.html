00108 - (void)_setStatusBarHidden:(BOOL)hidden withAnimation:(UIStatusBarAnimation)animation {
00109   [self _setStatusBarHidden:hidden withAnimation:animation];
00110 
00111   if (UIStatusBarAnimationNone == animation) {
00112     [NIOverview view].alpha = 1;
00113     [NIOverview view].hidden = hidden;
00114 
00115   } else if (UIStatusBarAnimationSlide == animation) {
00116     [NIOverview view].alpha = 1;
00117 
00118     CGRect frame = [NIOverview frame];
00119 
00120     [UIView beginAnimations:nil context:nil];
00121     [UIView setAnimationDuration:NIStatusBarAnimationDuration()];
00122     [UIView setAnimationCurve:NIStatusBarAnimationCurve()];
00123 
00124     [NIOverview view].frame = frame;
00125 
00126     [UIView commitAnimations];
00127     
00128   } else if (UIStatusBarAnimationFade == animation) {
00129     CGRect frame = [NIOverview frame];
00130     [NIOverview view].frame = frame;
00131 
00132     [UIView beginAnimations:nil context:nil];
00133     [UIView setAnimationDuration:NIStatusBarAnimationDuration()];
00134     [UIView setAnimationCurve:NIStatusBarAnimationCurve()];
00135     
00136     [NIOverview view].alpha = hidden ? 0 : 1;
00137     
00138     [UIView commitAnimations];
00139   }
00140 }
