00361     self.lastPathToNetworkImage = pathToNetworkImage;

00363     // We explicitly do not allow negative display sizes. Check the call stack to figure
00364     // out who is providing a negative display size. It's possible that displaySize is an
00365     // uninitialized CGSize structure.
00366     NIDASSERT(displaySize.width >= 0);
00367     NIDASSERT(displaySize.height >= 0);
00368 
00369     // If an invalid display size is provided, use the image view's frame instead.
00370     if (0 >= displaySize.width || 0 >= displaySize.height) {
00371       displaySize = self.frame.size;
00372     }

00375 
00376     // Check for file URLs.
00377     if ([pathToNetworkImage hasPrefix:@"/"]) {
00378       // If the url starts with / then it's likely a file URL, so treat it accordingly.
00379       url = [NSURL fileURLWithPath:pathToNetworkImage];
00380 
00381     } else {
00382       // Otherwise we assume it's a regular URL.
00383       url = [NSURL URLWithString:pathToNetworkImage];
00384     }
00385 
00386     // If the URL failed to be created, there's not much we can do here.
00387     if (nil == url) {
00388       return;
00389     }
00390 
00391     UIImage* image = nil;
00392 
00393     // Attempt to load the image from memory first.
00394     if (nil != self.imageMemoryCache) {
00395       NSString* cacheKey = [self cacheKeyForURL: url
00396                                       imageSize: displaySize
00397                                     contentMode: contentMode
00398                                    scaleOptions: self.scaleOptions];
00399       image = [self.imageMemoryCache objectWithName:cacheKey];
00400     }
00401 
00402     if (nil != image) {
00403       // We successfully loaded the image from memory.
00404       [self setImage:image];
00405 
00406       if ([self.delegate respondsToSelector:@selector(networkImageView:didLoadImage:)]) {
00407         [self.delegate networkImageView:self didLoadImage:self.image];
00408       }
00409 
00410     } else {
00411       // Unable to load the image from memory, fire off the load request (which will load
00412       // the image from the disk if possible and fall back to loading from the network).
00413 
00414       // NIHTTPImageRequest handles file urls by simply loading the image from the disk and firing
00415       // off the necessary delegate notifications. No network objects are created in the
00416       // image request thread when this happens.
00417

00419       [NIHTTPImageRequest requestWithURL: url
00420                               usingCache: self.imageDiskCache];
00421 
00422       [request setDelegate:self];
00423       [request setDidFinishSelector:@selector(requestDidFinish:)];
00424       [request setDidFailSelector:@selector(requestDidFail:)];
00425 
00426       [request setCacheStoragePolicy:self.cacheStoragePolicy];
00427 
00428       [request setImageCropRect:cropRect];

00430       [request setInterpolationQuality:self.interpolationQuality];

00434       }
