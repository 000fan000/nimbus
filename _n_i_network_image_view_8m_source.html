00131                    imageSize: (CGSize)imageSize
00132                  contentMode: (UIViewContentMode)contentMode
00133                 scaleOptions: (NINetworkImageViewScaleOptions)scaleOptions {
00134   NSString* cacheKey = [URL absoluteString];
00135 
00136   // Prefix cache key to create a namespace.
00137   if (nil != self.memoryCachePrefix) {
00138     cacheKey = [self.memoryCachePrefix stringByAppendingString:cacheKey];
00139   }
00140 
00141   // Append the size to the key. This allows us to differentiate cache keys by image dimension.
00142   // If the display size ever changes, we want to ensure that we're fetching the correct image
00143   // from the cache.
00144   if (self.sizeForDisplay) {
00145     cacheKey = [cacheKey stringByAppendingFormat:@"%@{%d,%d}",
00146                 NSStringFromCGSize(imageSize), contentMode, scaleOptions];
00147   }
00148 
00149   // The resulting cache key will look like:
00150   // (memoryCachePrefix)/path/to/image({width,height}{contentMode,cropImageForDisplay})
00151 
00152   return cacheKey;

00341 - (void)setPathToNetworkImage:(NSString *)pathToNetworkImage forDisplaySize:(CGSize)displaySize contentMode:(UIViewContentMode)contentMode cropRect:(CGRect)cropRect {
00342   [self cancelOperation];
00343 
00344   if (NIIsStringWithAnyText(pathToNetworkImage)) {
00345     self.lastPathToNetworkImage = pathToNetworkImage;
00346 
00347     // We explicitly do not allow negative display sizes. Check the call stack to figure
00348     // out who is providing a negative display size. It's possible that displaySize is an
00349     // uninitialized CGSize structure.
00350     NIDASSERT(displaySize.width >= 0);
00351     NIDASSERT(displaySize.height >= 0);
00352 
00353     // If an invalid display size is provided, use the image view's frame instead.
00354     if (0 >= displaySize.width || 0 >= displaySize.height) {
00355       displaySize = self.frame.size;
00356     }
00357 
00358     NSURL* url = nil;
00359 
00360     // Check for file URLs.
00361     if ([pathToNetworkImage hasPrefix:@"/"]) {
00362       // If the url starts with / then it's likely a file URL, so treat it accordingly.
00363       url = [NSURL fileURLWithPath:pathToNetworkImage];
00364 
00365     } else {
00366       // Otherwise we assume it's a regular URL.
00367       url = [NSURL URLWithString:pathToNetworkImage];
00368     }
00369 
00370     // If the URL failed to be created, there's not much we can do here.
00371     if (nil == url) {
00372       return;
00373     }
00374 
00375     UIImage* image = nil;
00376 
00377     // Attempt to load the image from memory first.
00378     if (nil != self.imageMemoryCache) {
00379       NSString* cacheKey = [self cacheKeyForURL: url
00380                                       imageSize: displaySize
00381                                     contentMode: contentMode
00382                                    scaleOptions: self.scaleOptions];
00383       image = [self.imageMemoryCache objectWithName:cacheKey];
00384     }
00385 
00386     if (nil != image) {
00387       // We successfully loaded the image from memory.
00388       [self setImage:image];
00389 
00390       if ([self.delegate respondsToSelector:@selector(networkImageView:didLoadImage:)]) {
00391         [self.delegate networkImageView:self didLoadImage:self.image];
00392       }
00393 
00394     } else {
00395       // Unable to load the image from memory, fire off the load request (which will load
00396       // the image from the disk if possible and fall back to loading from the network).
00397 
00398       // NIHTTPImageRequest handles file urls by simply loading the image from the disk and firing
00399       // off the necessary delegate notifications. No network objects are created in the
00400       // image request thread when this happens.
00401 
00402       NIHTTPImageRequest* request =
00403       [NIHTTPImageRequest requestWithURL: url
00404                               usingCache: self.imageDiskCache];
00405 
00406       [request setDelegate:self];
00407       [request setDidFinishSelector:@selector(requestDidFinish:)];
00408       [request setDidFailSelector:@selector(requestDidFail:)];
00409 
00410       [request setCacheStoragePolicy:self.cacheStoragePolicy];
00411 
00412       [request setImageCropRect:cropRect];
00413       [request setScaleOptions:self.scaleOptions];
00414       [request setInterpolationQuality:self.interpolationQuality];
00415       if (self.sizeForDisplay) {
00416         [request setImageDisplaySize:displaySize];
00417         [request setImageContentMode:contentMode];
00418       }
00419 
00420       self.operation = request;
00421 
00422       [self.networkOperationQueue addOperation:self.operation];
00423     }
00424   }
