00124                 scaleOptions: (NINetworkImageViewScaleOptions)scaleOptions {

00126 
00127   // Prefix cache key to create a namespace.
00128   if (nil != self.memoryCachePrefix) {
00129     cacheKey = [self.memoryCachePrefix stringByAppendingString:cacheKey];
00130   }
00131 
00132   // Append the size to the key. This allows us to differentiate cache keys by image dimension.
00133   // If the display size ever changes, we want to ensure that we're fetching the correct image
00134   // from the cache.

00136     cacheKey = [cacheKey stringByAppendingString:NSStringFromCGSize(imageSize)];
00137   }
00138 
00139   cacheKey = [cacheKey stringByAppendingFormat:@"{%d,%d}", contentMode, scaleOptions];

00142   // (memoryCachePrefix)/path/to/image({width,height}){contentMode,cropImageForDisplay}
