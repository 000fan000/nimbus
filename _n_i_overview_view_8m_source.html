00092 - (CGRect)frameForPagingScrollView {
00093   CGRect frame = self.bounds;
00094 
00095   // We make the paging scroll view a little bit wider on the side edges so that there
00096   // there is space between the pages when flipping through them.
00097   frame.origin.x -= self.pageHorizontalMargin;

00112 - (CGRect)frameForPageAtIndex:(NSInteger)index {
00113   // We have to use our paging scroll view's bounds, not frame, to calculate the page
00114   // placement. When the device is in landscape orientation, the frame will still be in
00115   // portrait because the pagingScrollView is the root view controller's view, so its
00116   // frame is in window coordinate space, which is never rotated. Its bounds, however,
00117   // will be in landscape because it has a rotation transform applied.
00118   CGRect bounds = _pagingScrollView.bounds;
00119   CGRect pageFrame = bounds;
00120 
00121   // We need to counter the extra spacing added to the paging scroll view in
00122   // frameForPagingScrollView:
00123   pageFrame.size.width -= self.pageHorizontalMargin * 2;
00124   pageFrame.origin.x = (bounds.size.width * index) + self.pageHorizontalMargin;
00125 
00126   return pageFrame;
00127 }
