00123 - (CGRect)frameForPageAtIndex:(NSInteger)pageIndex {
00124   // We have to use our paging scroll view's bounds, not frame, to calculate the page
00125   // placement. When the device is in landscape orientation, the frame will still be in
00126   // portrait because the pagingScrollView is the root view controller's view, so its
00127   // frame is in window coordinate space, which is never rotated. Its bounds, however,
00128   // will be in landscape because it has a rotation transform applied.
00129   CGRect bounds = _pagingScrollView.bounds;
00130   CGRect pageFrame = bounds;
00131 
00132   // We need to counter the extra spacing added to the paging scroll view in
00133   // frameForPagingScrollView:
00134   pageFrame.size.width -= self.pageHorizontalMargin * 2;
00135   pageFrame.origin.x = (bounds.size.width * pageIndex) + self.pageHorizontalMargin;
00136 
00137   return pageFrame;

00297 - (void)updateVisiblePages {
00298   NSInteger oldCenterPhotoIndex = self.centerPhotoIndex;
00299 
00300   NSRange visiblePageRange = [self visiblePageRange];
00301 
00302   _centerPhotoIndex = [self currentVisiblePageIndex];
00303 
00304   // Recycle no-longer-visible pages.
00305   for (NIPhotoScrollView* page in _visiblePages) {
00306     if (!NSLocationInRange(page.photoIndex, visiblePageRange)) {
00307       [_recycledPages addObject:page];
00308       [page removeFromSuperview];
00309 
00310       // Give the data source the opportunity to kill any asynchronous operations for this
00311       // now-recycled page.
00312       if ([_dataSource respondsToSelector:
00313            @selector(photoAlbumScrollView:stopLoadingPhotoAtIndex:)]) {
00314         [_dataSource photoAlbumScrollView: self
00315                   stopLoadingPhotoAtIndex: page.photoIndex];
00316       }
00317     }
00318   }
00319   [_visiblePages minusSet:_recycledPages];
00320 
00321   // Prioritize displaying the currently visible page.
00322   if (![self isDisplayingPageForIndex:_centerPhotoIndex]) {
00323     [self displayPageAtIndex:_centerPhotoIndex];
00324   }
00325 
00326   // Add missing pages.
00327   for (int pageIndex = visiblePageRange.location;
00328        pageIndex < NSMaxRange(visiblePageRange); ++pageIndex) {
00329     if (![self isDisplayingPageForIndex:pageIndex]) {
00330       [self displayPageAtIndex:pageIndex];
00331     }
00332   }
00333 
00334   if (oldCenterPhotoIndex != _centerPhotoIndex
00335       && [self.delegate respondsToSelector:@selector(photoAlbumScrollViewDidChangePages:)]) {
00336     [self.delegate photoAlbumScrollViewDidChangePages:self];
00337   }
