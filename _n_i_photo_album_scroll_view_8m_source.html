00121 - (CGRect)frameForPageAtIndex:(NSInteger)index {
00122   // We have to use our paging scroll view's bounds, not frame, to calculate the page
00123   // placement. When the device is in landscape orientation, the frame will still be in
00124   // portrait because the pagingScrollView is the root view controller's view, so its
00125   // frame is in window coordinate space, which is never rotated. Its bounds, however,
00126   // will be in landscape because it has a rotation transform applied.
00127   CGRect bounds = _pagingScrollView.bounds;
00128   CGRect pageFrame = bounds;
00129 
00130   // We need to counter the extra spacing added to the paging scroll view in
00131   // frameForPagingScrollView:
00132   pageFrame.size.width -= self.pageHorizontalMargin * 2;
00133   pageFrame.origin.x = (bounds.size.width * index) + self.pageHorizontalMargin;
00134 
00135   return pageFrame;
00136 }

00293 - (void)updateVisiblePages {
00294   NSInteger oldCenterPhotoIndex = self.currentCenterPhotoIndex;
00295 
00296   NSRange visiblePageRange = [self visiblePageRange];
00297 
00298   _currentCenterPhotoIndex = [self currentVisiblePageIndex];
00299 
00300   // Recycle no-longer-visible pages.
00301   for (NIPhotoScrollView* page in _visiblePages) {
00302     if (!NSLocationInRange(page.photoIndex, visiblePageRange)) {
00303       [_recycledPages addObject:page];
00304       [page removeFromSuperview];
00305 
00306       // Give the data source the opportunity to kill any asynchronous operations for this
00307       // now-recycled page.
00308       if ([_dataSource respondsToSelector:
00309            @selector(photoAlbumScrollView:stopLoadingPhotoAtIndex:)]) {
00310         [_dataSource photoAlbumScrollView: self
00311                   stopLoadingPhotoAtIndex: page.photoIndex];
00312       }
00313     }
00314   }
00315   [_visiblePages minusSet:_recycledPages];
00316 
00317   // Prioritize displaying the currently visible page.
00318   if (![self isDisplayingPageForIndex:_currentCenterPhotoIndex]) {
00319     [self displayPageAtIndex:_currentCenterPhotoIndex];
00320   }
00321 
00322   // Add missing pages.
00323   for (int index = visiblePageRange.location; index < NSMaxRange(visiblePageRange); ++index) {
00324     if (![self isDisplayingPageForIndex:index]) {
00325       [self displayPageAtIndex:index];
00326     }
00327   }
00328 
00329   if (oldCenterPhotoIndex != _currentCenterPhotoIndex
00330       && [self.delegate respondsToSelector:@selector(photoAlbumScrollViewDidChangePages:)]) {
00331     [self.delegate photoAlbumScrollViewDidChangePages:self];
00332   }
