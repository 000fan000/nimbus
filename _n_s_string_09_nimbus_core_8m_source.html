00081   while (![scanner isAtEnd]) {
00082     NSString* pairString = nil;
00083     [scanner scanUpToCharactersFromSet:delimiterSet intoString:&pairString];
00084     [scanner scanCharactersFromSet:delimiterSet intoString:NULL];

00087       NSString* key = [[kvPair objectAtIndex:0]
00088                        stringByReplacingPercentEscapesUsingEncoding:encoding];
00089       NSMutableArray* values = [pairs objectForKey:key];
00090       if (nil == values) {
00091         values = [NSMutableArray array];
00092         [pairs setObject:values forKey:key];
00093       }
00094       if (kvPair.count == 1) {
00095         [values addObject:[NSNull null]];
00096 
00097       } else if (kvPair.count == 2) {
00098         NSString* value = [[kvPair objectAtIndex:1]
00099                            stringByReplacingPercentEscapesUsingEncoding:encoding];
00100         [values addObject:value];
00101       }
00102     }
00103   }

00181 - (NSComparisonResult)versionStringCompare:(NSString *)other {
00182   NSArray *oneComponents = [self componentsSeparatedByString:@"a"];
00183   NSArray *twoComponents = [other componentsSeparatedByString:@"a"];
00184 
00185   // The parts before the "a"
00186   NSString *oneMain = [oneComponents objectAtIndex:0];
00187   NSString *twoMain = [twoComponents objectAtIndex:0];
00188 
00189   // If main parts are different, return that result, regardless of alpha part
00190   NSComparisonResult mainDiff;
00191   if ((mainDiff = [oneMain compare:twoMain]) != NSOrderedSame) {
00192     return mainDiff;
00193   }
00194 
00195   // At this point the main parts are the same; just deal with alpha stuff
00196   // If one has an alpha part and the other doesn't, the one without is newer
00197   if ([oneComponents count] < [twoComponents count]) {
00198     return NSOrderedDescending;
00199 
00200   } else if ([oneComponents count] > [twoComponents count]) {
00201     return NSOrderedAscending;
00202 
00203   } else if ([oneComponents count] == 1) {
00204     // Neither has an alpha part, and we know the main parts are the same
00205     return NSOrderedSame;
00206   }
00207 
00208   // At this point the main parts are the same and both have alpha parts. Compare the alpha parts
00209   // numerically. If it's not a valid number (including empty string) it's treated as zero.
00210   NSNumber *oneAlpha = [NSNumber numberWithInt:[[oneComponents objectAtIndex:1] intValue]];
00211   NSNumber *twoAlpha = [NSNumber numberWithInt:[[twoComponents objectAtIndex:1] intValue]];
00212   return [oneAlpha compare:twoAlpha];
00213 }
