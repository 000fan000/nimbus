00063   NSAutoreleasePool* pool = [[NSAutoreleasePool alloc] init];
00064   
00065   if ([self.url isFileURL]) {
00066     // Special case: load the image from disk without hitting the network.
00067 
00068     [self operationDidStart];
00069 
00070     NSError* dataReadError = nil;
00071 
00072     // The meat of the load-from-disk operation.
00073     NSString* filePath = [self.url path];
00074     NSMutableData* data = [NSMutableData dataWithContentsOfFile:filePath
00075                                                         options:0
00076                                                           error:&dataReadError];
00077 
00078     if (nil != dataReadError) {
00079       // This generally happens when the file path points to a file that doesn't exist.
00080       // dataReadError has the complete details.
00081       [self operationDidFailWithError:dataReadError];
00082 
00083     } else {
00084       self.data = data;
00085 
00086       // Notifies the delegates of the request completion.
00087       [self operationWillFinish];
00088       [self operationDidFinish];
00089     }
00090 
00091   } else {
00092     // Load the image from the network then.
00093     [self operationDidStart];
00094 
00095     NSURLRequest* request = [NSURLRequest requestWithURL:self.url
00096                                              cachePolicy:NSURLRequestUseProtocolCachePolicy
00097                                          timeoutInterval:self.timeout];
00098 
00099     NSError* networkError = nil;
00100     NSURLResponse* response = nil;
00101     NSData* data  = [NSURLConnection sendSynchronousRequest:request
00102                                           returningResponse:&response
00103                                                       error:&networkError];
00104 
00105     if (nil != networkError) {
00106       [self operationDidFailWithError:networkError];
00107 
00108     } else {
00109       self.data = data;
00110 
00111       [self operationWillFinish];
00112       [self operationDidFinish];
00113     }
00114   }
