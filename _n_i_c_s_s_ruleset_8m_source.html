00159 - (UIFont *)font {
00160   NIDASSERT([self hasFont]);
00161   
00162   if (_is.cached.Font) {
00163     return _font;
00164   }
00165 
00166   NSString* fontName = nil;
00167   CGFloat fontSize = [UIFont systemFontSize];
00168   BOOL fontIsBold = NO;
00169   BOOL fontIsItalic = NO;
00170   
00171   NSArray* values = [_ruleset objectForKey:kFontWeightKey];
00172   if (nil != values) {
00173     NIDASSERT([values count] == 1);
00174     fontIsBold = [[values objectAtIndex:0] isEqualToString:@"bold"];
00175   }
00176   
00177   values = [_ruleset objectForKey:kFontStyleKey];
00178   if (nil != values) {
00179     NIDASSERT([values count] == 1);
00180     fontIsItalic = [[values objectAtIndex:0] isEqualToString:@"italic"];
00181   }
00182 
00183   // There are two ways to set font size and family: font and font-size/font-family.
00184   // Newer definitions of these values should overwrite previous definitions so we must
00185   // respect ordering here.
00186   BOOL hasSetFontName = NO;
00187   BOOL hasSetFontSize = NO;
00188 
00189   NSArray* order = [_ruleset objectForKey:kPropertyOrderKey];
00190   for (NSString* name in [order reverseObjectEnumerator]) {
00191     if (!hasSetFontName && [name isEqualToString:kFontFamilyKey]) {
00192       values = [_ruleset objectForKey:name];
00193       NIDASSERT([values count] == 1); if ([values count] < 1) { continue; }
00194       fontName = [[values objectAtIndex:0] stringByTrimmingCharactersInSet:
00195                   [NSCharacterSet characterSetWithCharactersInString:@"\""]];
00196       hasSetFontName = YES;
00197 
00198     } else if (!hasSetFontSize && [name isEqualToString:kFontSizeKey]) {
00199       values = [_ruleset objectForKey:name];
00200       NIDASSERT([values count] == 1); if ([values count] < 1) { continue; }
00201       NSString* value = [values objectAtIndex:0];
00202       if ([value isEqualToString:@"default"]) {
00203         fontSize = [UIFont systemFontSize];
00204 
00205       } else {
00206         fontSize = [value floatValue];
00207       }
00208       hasSetFontSize = YES;
00209 
00210     } else if (!hasSetFontSize && !hasSetFontName && [name isEqualToString:kFontKey]) {
00211       values = [_ruleset objectForKey:name];
00212       NIDASSERT([values count] <= 2); if ([values count] < 1) { continue; }
00213 
00214       if ([values count] >= 1) {
00215         // Font size
00216         fontSize = [[values objectAtIndex:0] floatValue];
00217         hasSetFontSize = YES;
00218       }
00219       if ([values count] >= 2) {
00220         // Font name
00221         fontName = [[values objectAtIndex:1] stringByTrimmingCharactersInSet:
00222                     [NSCharacterSet characterSetWithCharactersInString:@"\""]];
00223         hasSetFontName = YES;
00224       }
00225     }
00226 
00227     if (hasSetFontName && hasSetFontSize) {
00228       // Once we've set all values then we can ignore any previous values.
00229       break;
00230     }
00231   }
00232   
00233   UIFont* font = nil;
00234   if (hasSetFontName) {
00235     // If you wish to set the weight and style for a non-standard font family then you will need
00236     // to set the font family to the given style manually.
00237     NIDASSERT(!fontIsItalic && !fontIsBold);
00238     font = [UIFont fontWithName:fontName size:fontSize];
00239 
00240   } else if (fontIsItalic && fontIsBold) {
00241     // There is no easy way to create a bold italic font using the exposed UIFont methods.
00242     // Please consider using the exact font name instead. E.g. font-name: Helvetica-BoldObliquei
00243     NIDASSERT(NO);
00244     font = [UIFont systemFontOfSize:fontSize];
00245 
00246   } else if (fontIsItalic) {
00247     font = [UIFont italicSystemFontOfSize:fontSize];
00248 
00249   } else if (fontIsBold) {
00250     font = [UIFont boldSystemFontOfSize:fontSize];
00251 
00252   } else {
00253     font = [UIFont systemFontOfSize:fontSize];
00254   }
00255 
00256   [_font release];
00257   _font = [font retain];
00258   _is.cached.Font = YES;
00259 
00260   return font;
00261 }

00490 - (void)cacheBorderValues {
00491   _borderWidth = 0;
00492   NI_RELEASE_SAFELY(_borderColor);
00493 
00494   NSArray* values = nil;
00495 
00496   // There are two ways to set border color and width: border and border-color/border-width.
00497   // Newer definitions of these values should overwrite previous definitions so we must
00498   // respect ordering here.
00499   BOOL hasSetBorderColor = NO;
00500   BOOL hasSetBorderWidth = NO;
00501 
00502   NSArray* order = [_ruleset objectForKey:kPropertyOrderKey];
00503   for (NSString* name in [order reverseObjectEnumerator]) {
00504     if (!hasSetBorderColor && [name isEqualToString:kBorderColorKey]) {
00505       values = [_ruleset objectForKey:name];
00506       _borderColor = [[[self class] colorFromCssValues:values
00507                                 numberOfConsumedTokens:nil] retain];
00508       hasSetBorderColor = YES;
00509 
00510     } else if (!hasSetBorderWidth && [name isEqualToString:kBorderWidthKey]) {
00511       values = [_ruleset objectForKey:name];
00512       NIDASSERT([values count] == 1); if ([values count] < 1) { continue; }
00513       _borderWidth = [[values objectAtIndex:0] floatValue];
00514       hasSetBorderWidth = YES;
00515 
00516     } else if (!hasSetBorderColor && !hasSetBorderWidth && [name isEqualToString:kBorderKey]) {
00517       values = [_ruleset objectForKey:name];
00518 
00519       if ([values count] >= 1) {
00520         // Border width
00521         _borderWidth = [[values objectAtIndex:0] floatValue];
00522         hasSetBorderWidth = YES;
00523       }
00524       if ([values count] >= 3) {
00525         // Border color
00526         _borderColor = [[[self class] colorFromCssValues:[values subarrayWithRange:NSMakeRange(2, [values count] - 2)]
00527                                   numberOfConsumedTokens:nil] retain];
00528         hasSetBorderColor = YES;
00529       }
00530     }
00531     
00532     if (hasSetBorderColor && hasSetBorderWidth) {
00533       // Once we've set all values then we can ignore any previous values.
00534       break;
00535     }
00536   }
00537   
00538   _is.cached.BorderColor = hasSetBorderColor;
00539   _is.cached.BorderWidth = hasSetBorderWidth;
00540 }
