00406 //
00407 // iPhone condition codes as reported by a 3GS running iPhone OS v3.0.
00408 // Airplane Mode turned on:  Reachability Flag Status: -- -------
00409 // WWAN Active:              Reachability Flag Status: WR -t-----
00410 // WWAN Connection required: Reachability Flag Status: WR ct-----
00411 //         WiFi turned on:   Reachability Flag Status: -R ------- Reachable.
00412 // Local   WiFi turned on:   Reachability Flag Status: -R xxxxxxd Reachable.
00413 //         WiFi turned on:   Reachability Flag Status: -R ct----- Connection down. (Non-intuitive, empirically determined answer.)
00414 const SCNetworkReachabilityFlags kConnectionDown =  kSCNetworkReachabilityFlagsConnectionRequired | 
00415                             kSCNetworkReachabilityFlagsTransientConnection;
00416 //         WiFi turned on:   Reachability Flag Status: -R ct-i--- Reachable but it will require user intervention (e.g. enter a WiFi password).
00417 //         WiFi turned on:   Reachability Flag Status: -R -t----- Reachable via VPN.
00418 //
00419 // In the below method, an 'x' in the flag status means I don't care about its value.
00420 //
00421 // This method differs from Apple's by testing explicitly for empirically observed values.
00422 // This gives me more confidence in it's correct behavior. Apple's code covers more cases 
00423 // than mine. My code covers the cases that occur.
00424 //
00425 - (NetworkStatus) networkStatusForFlags: (SCNetworkReachabilityFlags) flags {
00426   
00427   if (flags & kSCNetworkReachabilityFlagsReachable) {
00428     
00429     // Local WiFi -- Test derived from Apple's code: -localWiFiStatusForFlags:.
00430     if (self.key == kLocalWiFiConnection) {
00431 
00432       // Reachability Flag Status: xR xxxxxxd Reachable.
00433       return (flags & kSCNetworkReachabilityFlagsIsDirect) ? kReachableViaWiFi : kNotReachable;
00434 
00435     }
00436     
00437     // Observed WWAN Values:
00438     // WWAN Active:              Reachability Flag Status: WR -t-----
00439     // WWAN Connection required: Reachability Flag Status: WR ct-----
00440     //
00441     // Test Value: Reachability Flag Status: WR xxxxxxx
00442     if (flags & kSCNetworkReachabilityFlagsIsWWAN) { return kReachableViaWWAN; }
00443     
00444     // Clear moot bits.
00445     flags &= ~kSCNetworkReachabilityFlagsReachable;
00446     flags &= ~kSCNetworkReachabilityFlagsIsDirect;
00447     flags &= ~kSCNetworkReachabilityFlagsIsLocalAddress; // kInternetConnection is local.
00448     
00449     // Reachability Flag Status: -R ct---xx Connection down.
00450     if (flags == kConnectionDown) { return kNotReachable; }
00451     
00452     // Reachability Flag Status: -R -t---xx Reachable. WiFi + VPN(is up) (Thank you Ling Wang)
00453     if (flags & kSCNetworkReachabilityFlagsTransientConnection)  { return kReachableViaWiFi; }
00454       
00455     // Reachability Flag Status: -R -----xx Reachable.
00456     if (flags == 0) { return kReachableViaWiFi; }
00457     
00458     // Apple's code tests for dynamic connection types here. I don't. 
00459     // If a connection is required, regardless of whether it is on demand or not, it is a WiFi connection.
00460     // If you care whether a connection needs to be brought up,   use -isConnectionRequired.
00461     // If you care about whether user intervention is necessary,  use -isInterventionRequired.
00462     // If you care about dynamically establishing the connection, use -isConnectionIsOnDemand.
00463 
00464     // Reachability Flag Status: -R cxxxxxx Reachable.
00465     if (flags & kSCNetworkReachabilityFlagsConnectionRequired) { return kReachableViaWiFi; }
00466     
00467     // Required by the compiler. Should never get here. Default to not connected.
00468 #if (defined DEBUG && defined CLASS_DEBUG)
00469     NSAssert1(NO, @"Uncaught reachability test. Flags: %@", reachabilityFlags_(flags));
00470 #endif
00471     return kNotReachable;
00472 
00473     }
00474   
00475   // Reachability Flag Status: x- xxxxxxx
00476   return kNotReachable;
00477   
00478 } // networkStatusForFlags:
