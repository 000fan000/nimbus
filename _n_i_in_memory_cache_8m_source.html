00227 - (void)storeObject:(id)object withName:(NSString *)name expiresAfter:(NSDate *)expirationDate {
00228   // Don't store nil objects in the cache.
00229   if (nil == object) {
00230     return;
00231   }
00232 
00233   if (nil != expirationDate && [[NSDate date] timeIntervalSinceDate:expirationDate] >= 0) {
00234     // The object being stored is already expired so remove the object from the cache altogether.
00235     [self removeObjectWithName:name];
00236 
00237     // We're done here.
00238     return;
00239   }
00240   NIMemoryCacheInfo* info = [self cacheInfoForName:name];
00241 
00242   // Create a new cache entry.
00243   if (nil == info) {
00244     info = [[NIMemoryCacheInfo alloc] init];
00245     info.name = name;
00246   }
00247 
00248   // Store the object in the cache item.
00249   info.object = object;
00250 
00251   // Override any existing expiration date.
00252   info.expirationDate = expirationDate;

00321 - (void)reduceMemoryUsage {
00322   // Copy the cache map because it's likely that we're going to modify it.
00323   NSDictionary* cacheMap = [_cacheMap copy];
00324 
00325   // Iterate over the copied cache map (which will not be modified).

00443 - (void)didSetObject:(id)object withName:(NSString *)name {
00444   // Reduce the cache size after the object has been set in case the cache size is smaller
00445   // than the object that's being added and we need to remove this object right away. If we
00446   // try to reduce the cache size before the object's been set, we won't have anything to remove
00447   // and we'll get stuck in an infinite loop.
00448   if (self.maxNumberOfPixels > 0) {
00449     // Remove least recently used images until we satisfy our memory constraints.
00450     while (self.numberOfPixels > self.maxNumberOfPixels
00451            && [self.lruCacheObjects count] > 0) {
00452       NIMemoryCacheInfo* info = [self.lruCacheObjects firstObject];
00453       [self removeCacheInfoForName:info.name];
