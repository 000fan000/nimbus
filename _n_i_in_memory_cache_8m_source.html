00236     // The object being stored is already expired so remove the object from the cache altogether.

00481 - (void)didSetObject:(id)object withName:(NSString *)name {
00482   // Reduce the cache size after the object has been set in case the cache size is smaller
00483   // than the object that's being added and we need to remove this object right away. If we
00484   // try to reduce the cache size before the object's been set, we won't have anything to remove
00485   // and we'll get stuck in an infinite loop.
00486   if (self.maxNumberOfPixels > 0) {
00487     // Remove least recently used images until we satisfy our memory constraints.
00488     while (self.numberOfPixels > self.maxNumberOfPixels
00489            && [self.lruCacheObjects count] > 0) {
00490       NIMemoryCacheInfo* info = [self.lruCacheObjects firstObject];
00491       [self removeCacheInfoForName:info.name];
