00106 -(NSDate *)dateFromString:(NSString *)dateString
00107 {
00108         // We store our date formatter in the calling thread's dictionary
00109         // NSDateFormatter is not thread-safe, this approach ensures each formatter is only used on a single thread
00110         // This formatter can be reused many times in parsing a single response, so it would be expensive to keep creating new date formatters
00111         NSMutableDictionary *threadDict = [[NSThread currentThread] threadDictionary];
00112         NSDateFormatter *dateFormatter = [threadDict objectForKey:@"ASICloudFilesResponseDateFormatter"];
00113         if (dateFormatter == nil) {
00114                 dateFormatter = [[[NSDateFormatter alloc] init] autorelease];
00115                 [dateFormatter setLocale:[[[NSLocale alloc] initWithLocaleIdentifier:@"en_US_POSIX"] autorelease]];
00116                 // example: 2009-11-04T19:46:20.192723
00117                 [dateFormatter setDateFormat:@"yyyy-MM-dd'T'H:mm:ss.SSSSSS"];
00118                 [threadDict setObject:dateFormatter forKey:@"ASICloudFilesResponseDateFormatter"];
00119         }
00120         return [dateFormatter dateFromString:dateString];
