00249         // NSDateFormatter is not thread-safe, this approach ensures each formatter is only used on a single thread
00250         // This formatter can be reused 1000 times in parsing a single response, so it would be expensive to keep creating new date formatters
00251         NSMutableDictionary *threadDict = [[NSThread currentThread] threadDictionary];
00252         NSDateFormatter *dateFormatter = [threadDict objectForKey:@"ASIS3ResponseDateFormatter"];
00253         if (dateFormatter == nil) {

00255                 [dateFormatter setLocale:[[[NSLocale alloc] initWithLocaleIdentifier:@"en_US_POSIX"] autorelease]];

00257                 [dateFormatter setDateFormat:@"yyyy-MM-dd'T'HH:mm:ss'.000Z'"];
