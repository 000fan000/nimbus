00037 NSString* NIStringFromBytes(unsigned long long bytes) {
00038   static const void* sOrdersOfMagnitude[] = {
00039     @"bytes", @"KB", @"MB", @"GB"
00040   };
00041 
00042   // Determine what magnitude the number of bytes is by shifting off 10 bits at a time
00043   // (equivalent to dividing by 1024).
00044   NSInteger magnitude = 0;
00045   unsigned long long highbits = bytes;
00046   unsigned long long inverseBits = ~((unsigned long long)0x3FF);
00047   while ((highbits & inverseBits)
00048          && magnitude + 1 < (sizeof(sOrdersOfMagnitude) / sizeof(void *))) {
00049     // Shift off an order of magnitude.
00050     highbits >>= 10;
00051     magnitude++;
00052   }
00053 
00054   if (magnitude > 0) {
00055     unsigned long long dividend = 1024 << (magnitude - 1) * 10;
00056     double result = ((double)bytes / (double)(dividend));
00057     return [NSString stringWithFormat:@"%.2f %@",
00058             result,
00059             sOrdersOfMagnitude[magnitude]];
00060 
00061   } else {
00062     // We don't need to bother with dividing bytes.
00063     return [NSString stringWithFormat:@"%lld %@", bytes, sOrdersOfMagnitude[magnitude]];
00064   }
00065 }
