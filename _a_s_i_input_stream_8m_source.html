00048 - (NSInteger)read:(uint8_t *)buffer maxLength:(NSUInteger)len
00049 {
00050         [readLock lock];
00051         unsigned long toRead = len;
00052         if ([ASIHTTPRequest isBandwidthThrottled]) {
00053                 toRead = [ASIHTTPRequest maxUploadReadLength];
00054                 if (toRead > len) {
00055                         toRead = len;
00056                 } else if (toRead == 0) {
00057                         toRead = 1;
00058                 }
00059                 [request performThrottling];
00060         }
00061         [ASIHTTPRequest incrementBandwidthUsedInLastSecond:toRead];
00062         [readLock unlock];
00063         return [stream read:buffer maxLength:toRead];
