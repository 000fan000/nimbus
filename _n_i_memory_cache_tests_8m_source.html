00217 
00218   STAssertEquals([cache count], (NSUInteger)1, @"Cache should have one object.");
00219 
00220   // Accessing an expired object removes it from the cache entirely.
00221   STAssertNil([cache objectWithName:@"obj1"], @"Object should have expired.");
